// ─── MENU UTAMA ─────────────────────────────────────────────────────────────
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('Export PDF')
    .addItem('1. Update Formatting', 'updateFormatting')
    .addItem('2. Export ke PDF', 'exportActiveSheet')
    .addToUi();
}

// ─── 1. UPDATE FORMATTING ────────────────────────────────────────────────────

/**
 * Satu tombol untuk semua persiapan tampilan: tidak perlu select range manual.
 * Script otomatis mendeteksi baris header tabel (mencari kolom "Total" dan
 * "Harga Satuan"), lalu:
 *  - menambah/memperbarui 1 baris judul laporan di paling atas (nama sheet +
 *    tanggal hari ini, di-bold dan diberi warna)
 *  - merapikan kolom Total (dan Harga Satuan kalau ada) jadi format Rupiah
 *  - memberi conditional formatting (warna skala, makin tinggi makin gelap)
 *    pada kolom Total, supaya nilai tertinggi langsung kelihatan
 * Aman dijalankan berkali-kali — kalau judul sudah ada, script memperbarui
 * baris judul yang sama, bukan menambah baris baru.
 */
function updateFormatting() {
  var sheet = SpreadsheetApp.getActiveSheet();
  var ui = SpreadsheetApp.getUi();
  var dataRange = sheet.getDataRange();
  var values = dataRange.getValues();

  if (!values || values.length < 2) {
    ui.alert('Tidak ada data yang terdeteksi di sheet ini. Pastikan sheet aktif berisi tabel dengan baris header.');
    return;
  }

  // Kalau baris 1 sudah judul laporan dari run sebelumnya, header ada di baris 2
  var firstCellText = String(values[0][0] || '');
  var titleAlreadyExists = firstCellText.indexOf('— Laporan') !== -1;
  var headerRowNum = titleAlreadyExists ? 2 : 1;
  var headerValues = values[headerRowNum - 1].map(function (v) { return String(v).trim(); });

  // 1) Tambah / perbarui baris judul laporan
  var today = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'd MMMM yyyy');
  var titleText = SpreadsheetApp.getActiveSpreadsheet().getName() + ' — Laporan ' + today;
  var lastCol = Math.max(sheet.getLastColumn(), headerValues.length);
  if (!titleAlreadyExists) {
    sheet.insertRowBefore(1);
  }
  var titleRange = sheet.getRange(1, 1, 1, lastCol);
  titleRange.merge();
  titleRange.setValue(titleText);
  titleRange.setFontWeight('bold');
  titleRange.setFontSize(13);
  titleRange.setBackground('#6C47FF');
  titleRange.setFontColor('#FFFFFF');
  titleRange.setHorizontalAlignment('center');
  sheet.setRowHeight(1, 30);

  // 2) Rapikan kolom Total / Harga Satuan jadi format Rupiah + conditional formatting
  var totalColIdx = headerValues.indexOf('Total');
  var hargaColIdx = headerValues.indexOf('Harga Satuan');
  var firstDataRow = headerRowNum + 1 + (titleAlreadyExists ? 0 : 1); // baris 1 = judul, baris 2 = header, data mulai baris 3
  var lastRow = sheet.getLastRow();

  if (totalColIdx === -1 && hargaColIdx === -1) {
    ui.alert('Judul laporan ditambahkan, tapi kolom "Total" atau "Harga Satuan" tidak ditemukan di header — formatting angka & conditional formatting dilewati. Pastikan nama kolomnya persis "Total" / "Harga Satuan".');
    return;
  }

  if (lastRow >= firstDataRow) {
    [totalColIdx, hargaColIdx].forEach(function (idx) {
      if (idx === -1) return;
      sheet.getRange(firstDataRow, idx + 1, lastRow - firstDataRow + 1, 1).setNumberFormat('"Rp" #,##0');
    });

    if (totalColIdx !== -1) {
      var totalRange = sheet.getRange(firstDataRow, totalColIdx + 1, lastRow - firstDataRow + 1, 1);
      // Buang rule lama yang menyasar kolom Total (biar aman dijalankan berkali-kali)
      var rules = sheet.getConditionalFormatRules().filter(function (r) {
        return r.getRanges().every(function (rg) { return rg.getColumn() !== totalColIdx + 1; });
      });
      var rule = SpreadsheetApp.newConditionalFormatRule()
        .setGradientMaxpointWithValue('#6C47FF', SpreadsheetApp.InterpolationType.MAX, '')
        .setGradientMinpointWithValue('#FFFFFF', SpreadsheetApp.InterpolationType.MIN, '')
        .setRanges([totalRange])
        .build();
      rules.push(rule);
      sheet.setConditionalFormatRules(rules);
    }
  }

  ui.alert('Formatting selesai: judul laporan ditambahkan' + (totalColIdx !== -1 ? ', kolom Total dirapikan jadi Rupiah + conditional formatting' : '') + (hargaColIdx !== -1 ? ', kolom Harga Satuan dirapikan jadi Rupiah' : '') + '.');
}

// ─── 2. EXPORT PDF ────────────────────────────────────────────────────────────

/**
 * Export sheet yang sedang aktif jadi 1 file PDF rapi — termasuk judul dan
 * formatting yang sudah ditambahkan lewat "1. Update Formatting" di atas.
 */
function exportActiveSheet() {
  var sheet = SpreadsheetApp.getActiveSheet();
  var opts = { gid: sheet.getSheetId() };
  var fileName = SpreadsheetApp.getActiveSpreadsheet().getName() + ' - ' + sheet.getName() + '.pdf';
  runExport(opts, fileName, sheet.getName());
}

/**
 * Jalankan proses export: bangun URL export bawaan Google Sheets, ambil file
 * PDF-nya, simpan ke folder "Export PDF - Belajar Claude" di Drive-mu, lalu
 * tampilkan link untuk membukanya.
 */
function runExport(opts, fileName, label) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var ui = SpreadsheetApp.getUi();

  try {
    var url = buildExportUrl(ss, opts);
    var token = ScriptApp.getOAuthToken();
    var response = UrlFetchApp.fetch(url, {
      headers: { Authorization: 'Bearer ' + token }
    });
    var pdfBlob = response.getBlob().setName(fileName);

    var folder = getOrCreateExportFolder();
    var file = folder.createFile(pdfBlob);

    var html = HtmlService.createHtmlOutput(
      '<style>body{font-family:Arial,sans-serif;font-size:13px;line-height:1.6;padding:12px;}'
      + 'a.btn{display:inline-block;margin-top:10px;background:#6C47FF;color:#fff;text-decoration:none;'
      + 'padding:8px 16px;border-radius:8px;font-weight:600;}</style>'
      + '<h3 style="color:#6C47FF;margin-top:0;">PDF Siap ✓</h3>'
      + '<p>"' + label + '" berhasil di-export sebagai PDF dan disimpan di folder '
      + '<strong>Export PDF - Belajar Claude</strong> di Google Drive-mu.</p>'
      + '<a class="btn" href="' + file.getUrl() + '" target="_blank">Buka PDF</a>'
    ).setTitle('Export Selesai').setWidth(420).setHeight(220);
    ui.showModelessDialog(html, 'Export Selesai');
  } catch (e) {
    ui.alert('Gagal export PDF: ' + e.message);
  }
}

/**
 * Bangun URL export bawaan Google Sheets dengan pengaturan rapi: tanpa
 * gridlines, fit ke lebar halaman, ukuran A4, potrait, margin rapi.
 */
function buildExportUrl(ss, opts) {
  var base = 'https://docs.google.com/spreadsheets/d/' + ss.getId() + '/export?';
  var params = {
    format: 'pdf',
    size: 'A4',
    portrait: 'true',
    fitw: 'true',
    gridlines: 'false',
    printtitle: 'false',
    sheetnames: 'false',
    pagenum: 'CENTER',
    top_margin: '0.5',
    bottom_margin: '0.5',
    left_margin: '0.5',
    right_margin: '0.5',
    horizontal_alignment: 'CENTER'
  };
  for (var k in opts) params[k] = opts[k];

  var query = Object.keys(params).map(function (k) {
    return k + '=' + encodeURIComponent(params[k]);
  }).join('&');
  return base + query;
}

/**
 * Cari atau buat folder "Export PDF - Belajar Claude" di root Drive supaya
 * hasil export tidak berserakan — semua PDF dari script ini tersimpan rapi
 * di 1 tempat.
 */
function getOrCreateExportFolder() {
  var name = 'Export PDF - Belajar Claude';
  var folders = DriveApp.getFoldersByName(name);
  if (folders.hasNext()) return folders.next();
  return DriveApp.createFolder(name);
}
